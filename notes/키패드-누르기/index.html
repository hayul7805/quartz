    def solution(numbers, hand):
    pad = {
        1:(0,0), 2:(0,1), 3:(0,2),
        4:(1,0), 5:(1,1), 6:(1,2),
        7:(2,0), 8:(2,1), 9:(2,2),
        '*':(3,0), 0:(3,1), '#':(3,2)
    }
    L_prev_key = pad['*']
    R_prev_key = pad['#']
    answer = ''

    for key in numbers:
        if key in [1,4,7]:
            answer += 'L'
            L_prev_key = pad[key]
        elif key in [3,6,9]:
            answer += 'R'
            R_prev_key = pad[key]
        else:
            L_distance = abs(L_prev_key[0] - pad[key][0]) + abs(L_prev_key[1] - pad[key][1])
            R_distance = abs(R_prev_key[0] - pad[key][0]) + abs(R_prev_key[1] - pad[key][1])
            if L_distance < R_distance:
                answer += 'L'
                L_prev_key = pad[key]
            elif L_distance > R_distance:
                answer += 'R'
                R_prev_key = pad[key]
            else:
                if hand == 'right':
                    answer += 'R'
                    R_prev_key = pad[key]
                else:
                    answer += 'L'
                    L_prev_key = pad[key]
    return answer
