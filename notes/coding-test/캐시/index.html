from collections import deque

def solution(cacheSize, cities):
    answer = 0
    lru = deque([])

    for city in cities:

        city = city.lower()

        if city not in lru:
            if cacheSize == 0:
                answer += 5
                continue
            if len(lru) < cacheSize:
                lru.append(city)
            else:
                if len(lru) != 0:
                    lru.popleft()
                lru.append(city)
            answer += 5
        else:
            lru.remove(city)
            lru.append(city)
            answer +=1

    return answer
