def solution(n, computers):
    answer = 1
    visited = [False] * n
    start_node = 0
    
    while True:
        visited = dfs(start_node, computers, visited)
        
        if False in visited:
            start_node = visited.index(False)
            answer += 1
        else:
            break
    
    return answer
